//9. vsprintf - format output of a stdarg argument list

#include<stdio.h>
#include<stdarg.h>

char buffer[50];

int print(char *n, ...)
{
    va_list ptr;
    va_start(ptr,n);
    int res = vsprintf(buffer,n,ptr);
    va_end(ptr);
    return res;
}

int main()
{
    print("Arguments %d %s",100,"Hello");
    printf("%s",buffer);
    
    return 0;
}
